## Example

```tsx
import { useState } from 'react';
import { Window } from 'uzumaki';
import { render } from 'uzumaki-react';

const window = new Window('main', { width: 900, height: 620, title: 'My App' });

function App() {
  const [count, setCount] = useState(0);
  return (
    <view
      display="flex"
      justify="center"
      items="center"
      w="full"
      h="full"
      gap={16}
    >
      <text fontSize={28}>Count: {count}</text>
      <button
        onClick={() => setCount(count + 1)}
        px={18}
        py={10}
        rounded={8}
        bg="#27272a"
      >
        <text>Increment</text>
      </button>
    </view>
  );
}

render(window, <App />);
```
